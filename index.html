// TimberTrack — paste in Apps Script, deploy as Web App
const SS_ID = 'YOUR_SPREADSHEET_ID';

function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.tryLock(10000);
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SS_ID);
    const logSh = ss.getSheetByName('LOG_DATA');
    const sumSh = ss.getSheetByName('SUMMARY');
    const h = data.header;
    data.logs.forEach(r => logSh.appendRow([
      h.container,h.supplier,h.product,h.country,h.date,h.quality,h.incharge,
      r.no,r.le1,r.l,r.g1,r.g2,r.cbm1,r.cbm2
    ]));
    const s = data.summary;
    const row = [h.container,h.supplier,h.product,h.country,h.date,h.quality,
      s.pcs,s.totalCbm1,s.totalCft1,s.avgCbm1,s.totalCbm2,s.totalCft2,s.avgCbm2];
    const ex = findRow(sumSh, h.container);
    if (ex > 0) sumSh.getRange(ex,1,1,row.length).setValues([row]);
    else sumSh.appendRow(row);
    return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
      .setMimeType(ContentService.MimeType.JSON);
  } finally { lock.releaseLock(); }
}

function doGet() {
  const rows = SpreadsheetApp.openById(SS_ID)
    .getSheetByName('SUMMARY').getDataRange().getValues();
  const hd = rows[0];
  return ContentService.createTextOutput(JSON.stringify({
    status:'ok',
    data: rows.slice(1).map(r => Object.fromEntries(hd.map((k,i)=>[k,r[i]])))
  })).setMimeType(ContentService.MimeType.JSON);
}

function findRow(sh, id) {
  const v = sh.getRange(1,1,sh.getLastRow(),1).getValues();
  for (let i=1; i<v.length; i++) if(v[i][0]==id) return i+1;
  return -1;
}
